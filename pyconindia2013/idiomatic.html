# Print 20 random prime numbers < 1000

# Solution A
def is_prime(n):
    """ Is the number 'n' prime ? """        
    prime = True
    for i in range(2,int(pow(n,0.5))+1):
        if n % i==0: 
            prime = False
            break
    return prime

count, numbers = 0, []

while count<20:
    num = random.randrange(1,1000)
    if num not in numbers and is_prime(num):    
        numbers.append(num)
        count += 1

Transposing a Matrix (list of lists)

# Solution A
from random import randrange

class MatrixA(object):       
    def __init__(self,m,n):
        self.m, self.n = m,n
        self.rows=[[randrange(1,10) for j in range(n)] for i in range(m)]

    def __str__(self):
        s='\n'.join([' '.join([str(i) for i in r]) for r in self.rows])
        return s + '\n'

    def transpose(self):
        """ Transpose a matrix """
        rows2 = [[0]*self.m for i in range(self.n)]
        for x in range(self.m):
            for y in range(self.n):
                rows2[y][x] = self.rows[x][y]

        self.rows = rows2
        # Swap m and n
        tmp = self.m
        self.m = self.n
        self.n = tmp

Filter data from iterable using (multiple) conditions

# Solution A

def find_passed(marks_dict,cutoffs):
    """ Find students who passed for each subject """

    # Using a dictionary and two for loops
    pass_dict = {subject:[] for subject in cutoffs}

    for subject, cutoff in cutoffs.iteritems():
        for student, student_marks in marks_dict.iteritems():
            if student_marks[subject] >= cutoff:
               pass_dict[subject].append(student)

    return pass_dict

21/53

Filter data from iterable using (multiple) conditions

# Solution B

# Solved idiomatically by using itertools.compress which allows 
# "selecting" items from an iterable using a second iterable of 
# boolean "selectors".

def find_passed(marks_dict, cutoffs):
    """ Find students who passed for each subject """

    pass_dict = collections.defaultdict(list)

    # two dicts to single dictionary using 'dict comprehension'
    selectors = {(subject, student) :cutoffs[subject]<=val[subject] for 
                  subject in cutoffs for student,val in marks_dict.items()}
    # Select students who passed using itertools.compress
    passd = list(itertools.compress(selectors.keys(), selectors.values()))

    # Create dictionary using defaultdict(list)
    for subject,student in passd:
        pass_dict[subject].append(student)

    return pass_dict

Problem of Code Overhead

Look at the following code.

def add(x,y):
    """ Add 2 numbers """
    safe = (type(x) in (int, float)) and (type(y) in (int, float))
    if safe: return x + y
    else: raise TypeError, "Wrong type!"

def add(x,y):
    """ Multiple 2 numbers """
    safe = (type(x) in (int, float)) and (type(y) in (int, float))
    if safe: return x * y
    else: raise TypeError, "Wrong type!"

The type checks are basically management code which are common to both functions. Such code overhead is better moved to decorators at one single place.

Management code using Decorators

def isNumber(f):
    def wrapper(*args):
        if not all(type(item) in (int, float) for item in args):
            raise TypeError,'Wrong type!'
        return f(*args)
    return wrapper

@isNumber
def add(x,y):
    return x+y

@isNumber
def mul(x,y):
    return x*y

Caching Decorators

A common use of decorators is to cache frequently required information. Here is a simple example.

def cache(f):
    results = {}
    def wrapper(*args):
        if args in results:
            print 'From cache'
            return results[args]
        res = f(*args)
        results[args] = res
        return res
return wrapper

@cache
def factorial(n):
    return reduce(lambda x,y: x*y, range(1,n+1))
>>> factorial(5)
120
>>> factorial(10)
3628800
>>> factorial(5)
From cache
120
>>> factorial(10)

Factoring out local contexts with "with"

There are many cases in programming where you need to call some methods when you enter and exit out of the context of a statement. For example,

# Locking a critical section
# Solution A
lock = threading.Lock()
# Entering critical section
lock.acquire()
try:
    # Critical section code
    my_critical_section()
finally:
    # Exiting critical section      
    lock.release()

The with statement allows you to factor out these contexts. 

# Solution B
with threading.Lock(): # lock acquired at this point
    my_critical_section()
# lock released at this point

You can write your own context managers by using the contextlib module that can be used with the with statement.

Making sure an child text block is always printed nicely indented relative to its parent block

from contextlib import contextmanager

spaces = 0
@contextmanager
def indent():
    global spaces
    spaces += 1
    print '\t'*spaces,
    yield
    spaces -= 1

for i in range(2,5):
    with indent(): print 'This is number',i
        # Indent more
        with indent(): print 'This is its square',i*i

    This is its square 16

Context manager classes

You can also create classes that implement context management semantics by implementing special __enter__ and __exit__ methods.

class Indent(object):
    def __init__(self):
        self.spaces = 0

    def __enter__(self):
        self.spaces += 1
        print '\t'*self.spaces,

    def __exit__(self, type, value, traceback):
        self.spaces -= 1
